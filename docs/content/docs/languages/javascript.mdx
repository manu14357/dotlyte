## Installation

```bash
# npm
npm install dotlyte

# pnpm
pnpm add dotlyte

# yarn
yarn add dotlyte
```

### Requirements

- Node.js **18+** (ESM)
- TypeScript **5.0+** (optional but recommended)

## Quick Start

```typescript
import { load } from "dotlyte";

const config = load();

// Dot-notation access (via Proxy)
console.log(config.database.host); // "localhost"
console.log(config.database.port); // 5432
```

## API Reference

### `load(options?)`

```typescript
import { load } from "dotlyte";

// Basic
const config = load();

// With options
const config = load({
  files: [".env", "config.yaml"],
  prefix: "APP_",
  defaults: { port: 3000 },
  env: "production",
});
```

**Returns:** `Config` object.

### `LoadOptions`

```typescript
import type { LoadOptions } from "dotlyte";
```

| Field | Type | Default | Description |
|---|---|---|---|
| `files` | `string[]` | — | Explicit config file paths |
| `prefix` | `string` | — | Env var prefix to strip (`APP_DB_HOST` → `db.host`) |
| `defaults` | `Record<string, unknown>` | — | Default values (lowest priority) |
| `sources` | `string[]` | — | Custom source order (`"env"`, `"dotenv"`, `"yaml"`, `"json"`, `"toml"`, `"defaults"`) |
| `env` | `string` | — | Environment name (loads `.env.{env}`, `config.{env}.yaml`) |
| `schema` | `DotlyteSchema` | — | Schema rules for load-time validation |
| `strict` | `boolean` | `false` | Reject unknown keys not in schema |
| `plugins` | `Source[]` | — | Custom source plugins |
| `watch` | `boolean` | `false` | Enable file watching / hot-reload |
| `debounceMs` | `number` | `100` | Watcher debounce interval (ms) |
| `interpolate` | `boolean` | `true` | Enable `${VAR}` interpolation |
| `override` | `boolean` | `false` | `.env` files override OS env vars |
| `debug` | `boolean` | `false` | Debug logging of source resolution |
| `findUp` | `boolean` | `false` | Walk up parent dirs for config files |
| `rootMarkers` | `string[]` | `[".git", "package.json", ...]` | Project-root markers for `findUp` |
| `cwd` | `string` | `process.cwd()` | Base directory for file resolution |

### `Config`

#### `config.get(key, defaultValue?)`

```typescript
const port = config.get("server.port", 8080);     // number
const name = config.get("app.name", "myapp");      // string
const missing = config.get("nope");                // undefined
```

#### `config.require(key)`

```typescript
import { MissingRequiredKeyError } from "dotlyte";

try {
  const dbUrl = config.require("database_url");
} catch (e) {
  if (e instanceof MissingRequiredKeyError) {
    console.error(`Missing: ${e.key}`);
    console.error(`Checked: ${e.sourcesChecked}`);
  }
}
```

#### `config.requireKeys(...keys)`

Batch-require multiple keys. Returns `Record<string, unknown>`:

```typescript
const { host, port, name } = config.requireKeys("db.host", "db.port", "db.name");
```

#### `config.has(key)`

```typescript
if (config.has("redis.host")) {
  setupRedis(config.require("redis.host"));
}
```

#### `config.scope(prefix)`

Create a scoped sub-config:

```typescript
const db = config.scope("database");
console.log(db.get("host")); // same as config.get("database.host")
```

#### Dot-Notation Access

Config objects support dynamic property access via Proxy:

```typescript
config.database.host       // "localhost"
config.server.port         // 8080
```

#### Serialization & Export

```typescript
config.keys();              // Flat dot-notation keys
config.toFlatMap();         // { "db.host": "localhost", "db.port": 5432, ... }
config.toObject();          // Deep nested object (mutable copy)
config.toObjectRedacted();  // Sensitive values → "***"
config.toJSON();            // Same as toObjectRedacted() (used by JSON.stringify)
config.toString();          // Human-readable with redaction
config.writeTo("out.json"); // Write to file ("json" | "env" | "yaml" | "toml")
```

#### Typed Access

```typescript
const port = config.getTyped<number>("port", 3000);
const all = config.as<MyConfigInterface>();
```

#### Schema Validation

```typescript
import type { DotlyteSchema } from "dotlyte";

const schema: DotlyteSchema = {
  port: { type: "integer", required: true, min: 1, max: 65535 },
  host: { type: "string", required: true },
};

const violations = config.validate(schema);
config.assertValid(schema); // throws ValidationError
```

#### File Watching (Hot-Reload)

When `watch: true` is passed to `load()`:

```typescript
const config = load({ watch: true, debounceMs: 200 });

config.onChange((newConfig) => console.log("Config changed!"));
config.onKeyChange("database.host", (oldVal, newVal) => {
  console.log(`${oldVal} → ${newVal}`);
});
config.onError((err) => console.error("Watch error:", err));

// When done:
config.close();
```

### Variable Interpolation

Values can reference other config keys and env vars:

```bash
# In .env:
BASE_URL=https://api.example.com
API_ENDPOINT=${BASE_URL}/v1
GREETING=${NAME:-World}       # default if NAME is missing
REQUIRED=${SECRET:?must set}  # error if SECRET is missing
LITERAL=$$NOT_INTERPOLATED    # escape with $$
```

```typescript
import { interpolate } from "dotlyte";

// Also available as standalone function
const result = interpolate(template, variables);
```

### Encryption (AES-256-GCM)

```typescript
import {
  generateKey, encryptValue, decryptValue,
  encryptFile, decryptFile
} from "dotlyte";

const key = generateKey();   // 32-byte hex key

// Value-level
const encrypted = encryptValue("s3cret", key);  // "ENC[...]"
const plain = decryptValue(encrypted, key);       // "s3cret"

// File-level
encryptFile(".env", ".env.enc", key);
decryptFile(".env.enc", ".env.dec", key);
```

Values wrapped in `ENC[...]` are auto-decrypted during `load()`.

### Sensitive Value Masking

```typescript
import { REDACTED, redactObject, buildSensitiveSet } from "dotlyte";

config.toObjectRedacted();
// { "db.host": "localhost", "db.password": "***" }
```

### Error Types

All errors extend `DotlyteError` (which extends `Error`):

| Error Class | Code | Extra Properties |
|---|---|---|
| `DotlyteError` | `DOTLYTE_ERROR` | `key`, `code` |
| `MissingRequiredKeyError` | `MISSING_REQUIRED_KEY` | `sourcesChecked` |
| `ParseError` | `PARSE_ERROR` | `filePath` |
| `FileError` | `FILE_NOT_FOUND` | `filePath` |
| `ValidationError` | `VALIDATION_ERROR` | `violations` |
| `InterpolationError` | `INTERPOLATION_ERROR` | — |
| `DecryptionError` | `DECRYPTION_ERROR` | `filePath` |

```typescript
import { DotlyteError, MissingRequiredKeyError } from "dotlyte";

try {
  config.require("missing");
} catch (e) {
  if (e instanceof MissingRequiredKeyError) {
    console.error(e.code);            // "MISSING_REQUIRED_KEY"
    console.error(e.key);             // "missing"
    console.error(e.sourcesChecked);  // [".env", "config.yaml", ...]
  }
}
```

### Custom Source Plugin

```typescript
import type { Source } from "dotlyte";

const mySource: Source = {
  name: "my-source",
  parse() {
    return { custom: "value" };
  },
};

const config = load({ plugins: [mySource] });
```

## TypeScript Support

DOTLYTE ships with full TypeScript declarations. Both ESM and CJS outputs are generated via `tsup`:

```typescript
import { load, Config } from "dotlyte";
import type { LoadOptions, DotlyteSchema, SchemaRule } from "dotlyte";

const config: Config = load();
```

## Patterns

### Express.js

```typescript
import express from "express";
import { load } from "dotlyte";

const config = load({ prefix: "APP_", env: process.env.NODE_ENV });

const app = express();
app.listen(config.get("port", 3000), () => {
  console.log(`Server running on port ${config.get("port", 3000)}`);
});
```

### Next.js

```typescript
// next.config.ts
import { load } from "dotlyte";

const config = load({ env: process.env.NODE_ENV });

export default {
  env: {
    API_URL: config.require("api_url"),
    PUBLIC_KEY: config.get("public_key", ""),
  },
};
```

### Fastify

```typescript
import Fastify from "fastify";
import { load } from "dotlyte";

const config = load({ prefix: "SERVER_" });
const fastify = Fastify({
  logger: config.get("log_enabled", true),
});

fastify.listen({ port: config.get("port", 3000) });
```

## What's New in v0.1.2

### Typed Config

```typescript
import { createTypedConfig } from "dotlyte/typed";

const config = createTypedConfig({
  PORT: { type: "integer", required: true, min: 1, max: 65535 },
  DEBUG: { type: "boolean", default: false },
  LOG_LEVEL: { type: "string", enum: ["debug", "info", "warn", "error"] },
  API_KEY: { type: "string", required: true, sensitive: true },
});
```

### Boundary Config

```typescript
import { createBoundaryConfig } from "dotlyte/boundary";

const config = createBoundaryConfig({
  serverKeys: ["DATABASE_URL", "API_SECRET"],
  clientKeys: ["APP_NAME", "THEME"],
});

const server = config.server();
const client = config.client(); // Cannot access server keys
```

### Enhanced Encryption & Masking

```typescript
import { Encryption, Masking } from "dotlyte";

// Key rotation
const enc = new Encryption("my-secret-key-32-chars-long!!!!!");
const encrypted = enc.encrypt("secret");

// Pattern-based masking with audit
const masking = new Masking({
  keys: ["API_KEY"],
  patterns: [/sk-[a-zA-Z0-9]+/],
});
```

### Workspace & CLI

```typescript
import { Workspace } from "dotlyte/workspace";

const ws = Workspace.detect();
const config = ws.loadPackage("api");
```
